130:      * @ignore
131:      */
132:     protected function adjust(Map $fields)
133:     {
134:         // 年と月の不整合を調整します.
135:         $this->adjustMonth($fields);
136:         
137:         $year  = $fields->get(self::$YEAR);
138:         $month = $fields->get(self::$MONTH);
139:         $date  = $fields->get(self::$DATE);
140:         
141:         // 日が1より小さい場合は、月の繰り下げ調整をします.
142:         if ($date < 1) {
143:             while ($date < 1) {
144:                 $month --;
145:                 if ($month === 0) {
146:                     $month = 12;
147:                     $year --;
148:                 }
149:                 $fields->put(self::$YEAR,  $year);
150:                 $fields->put(self::$MONTH, $month);
151:                 $date += self::getDateCountOf($year, $month);
152:             }
153:             $fields->put(self::$DATE, $date);
154:             $this->adjust($fields);
155:             return;
156:         }
157:         
158:         // 日がこの月の日数より大きい場合は、月の繰上げ調整をします.
159:         $dateCount = self::getDateCountOf($year, $month);
160:         if ($dateCount < $date) {
161:             while ($dateCount < $date) {
162:                 $date -= $dateCount;
163:                 $month ++;
164:                 if ($month === 13) {
165:                     $month = 1;
166:                     $year ++;
167:                 }
168:                 $fields->put(self::$YEAR,  $year);
169:                 $fields->put(self::$MONTH, $month);
170:                 $dateCount = self::getDateCountOf($year, $month);
171:             }
172:             $fields->put(self::$DATE, $date);
173:             $this->adjust($fields);
174:             return;
175:         }
176:     }
