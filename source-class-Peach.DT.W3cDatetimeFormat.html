 30: /**
 31:  * {@link http://www.w3.org/TR/NOTE-datetime W3CDTF} と時間オブジェクトの相互変換を行うフォーマットです.
 32:  * 本来の W3CDTF は日付と時刻の間に "T" が入りますが (例: "YYYY-MM-DDThh:mm:ss")
 33:  * このクラスは SQL などで用いられる慣用表現 (例: "YYYY-MM-DD hh:mm:ss") のパースもできます.
 34:  * 
 35:  * parse メソッドは, 文字列の先頭が W3CDTF の書式にマッチするかどうかで妥当性をチェックします.
 36:  * "2012-05-21T07:30:00" のような文字列は
 37:  * parseDate, parseDatetime, parseTimestamp のすべてのメソッドで受け付けますが,
 38:  * "2012-05-21" のような文字列は parseDate だけしか受け付けません.
 39:  * 
 40:  * このクラスはタイムゾーンを適切に扱うことが出来ます.
 41:  * parse 系メソッドでは, 時間オブジェクトに変換する際に内部のタイムゾーンに合わせて時刻を調整します.
 42:  * 例えばシステム時刻が UTC+9 として, GMT の時間文字列 "2012-05-20T22:30Z" を parse した場合
 43:  * 9 時間進ませた時間オブジェクト (new Datetime(2012, 5, 21, 7, 30) に等価) を生成します.
 44:  * 
 45:  * 時間オブジェクトを書式化する際は, 出力用の時差に合わせた文字列を生成します.
 46:  * 例えば内部のタイムゾーンが UTC+9, 出力用のタイムゾーンが UTC+5 に設定されているものとします.
 47:  * この場合, 内部のタイムゾーンと出力用のタイムゾーンの時差が -4 時間となるため,
 48:  * new Datetime(2012, 5, 21, 7, 30) の時間オブジェクトを書式化した結果は
 49:  * "2012-05-21T03:30+5:00" となります.
 50:  */
 51: class W3cDatetimeFormat implements Format
 52: {
 53:     /**
 54:      * タイムゾーンを扱うのみ TRUE となります
 55:      * @var bool
 56:      */
 57:     private $usingTz;
 58:     
 59:     /**
 60:      * システム時刻の時差です (単位は分)
 61:      * @var int
 62:      */
 63:     private $internalOffset;
 64:     
 65:     /**
 66:      * フォーマットの時差です (単位は分)
 67:      * @var int
 68:      */
 69:     private $externalOffset;
 70:     
 71:     /**
 72:      * 日付文字列のパターンです.
 73:      * "YYYY-MM-DD" のような形式にマッチします.
 74:      * 
 75:      * @var string
 76:      */
 77:     private static $datePattern = "[0-9]{4}[^0-9][0-9]{2}[^0-9][0-9]{2}";
 78:     
 79:     /**
 80:      * タイムゾーンの文字列のパターンです. 以下の形式にマッチします.
 81:      * 
 82:      * - "Z"
 83:      * - "-hh:mm"
 84:      * - "+hh:mm"
 85:      * - 空文字列
 86:      * 
 87:      * @var string
 88:      */
 89:     private static $timeZonePattern = "(Z|[\\-\\+][0-9]{2}:[0-9]{2})?";
 90:     
 91:     /**
 92:      * 指定されたタイムゾーンの条件で, 新しいフォーマットを構築します.
 93:      * タイムゾーンを扱わない場合は, コンストラクタの代わりに
 94:      * {@link W3cDatetimeFormat::getInstance} を使って下さい.
 95:      * 
 96:      * 引数 $externalOffset を指定することで, 
 97:      * 書式化する際のタイムゾーンの値を設定することが出来ます.
 98:      * (デフォルトはシステム時刻のタイムゾーンとなります)
 99:      * さらに $internalOffset を指定することで,
100:      * システム時刻のタイムゾーンとして任意の値を指定することも出来ます.
101:      * 
102:      * $externalOffset と $internalOffset を省略した場合, デフォルト値としてシステム時刻の設定
103:      * ({@link Util::getTimeZoneOffset}) と同じ値が指定されます.
104:      * UTC+1 以上の時差を設定する場合は負の値,
105:      * UTC-1 以下の時差を設定する場合は正の値を指定してください.
106:      * 
107:      * 以下に, $externalOffset と $internalOffset の組み合わせによる動作例を示します.
108:      * 
109:      * <code>
110:      * // 書式・システム時刻共に UTC+9
111:      * $f1 = new W3cDatetimeFormat(-540, -540);
112:      * 
113:      * // 書式は GMT, システム時刻は UTC+9
114:      * $f2 = new W3cDatetimeFormat(0,    -540);
115:      * 
116:      * // 書式は UTC+9, システム時刻は GMT
117:      * $f3 = new W3cDatetimeFormat(-540,    0);
118:      * 
119:      * // 2012-05-21T07:30 の時間オブジェクトを生成
120:      * $d = new Datetime(2012, 5, 21, 7, 30);
121:      * var_dump($d->format($f1)); // "2012-05-21T07:30+09:00"
122:      * var_dump($d->format($f2)); // "2012-05-20T22:30Z"      (UTC+9 の時刻を GMT として書式化)
123:      * var_dump($d->format($f3)); // "2012-05-21T16:30+09:00" (GMT の時刻を UTC+9 として書式化)
124:      * </code>
125:      * 
126:      * @param int  $externalOffset フォーマットの時差 (単位は分, 省略した場合はシステム設定の値を使用)
127:      * @param int  $internalOffset システム時刻の時差 (単位は分, 省略した場合はシステム設定の値を使用)
128:      */
129:     public function __construct($externalOffset = null, $internalOffset = null)
130:     {
131:         if ($externalOffset === false) {
132:             // @codeCoverageIgnoreStart
133:             $this->usingTz        = false;
134:             $this->externalOffset = null;
135:             $this->internalOffset = null;
136:             // @codeCoverageIgnoreEnd
137:         } else {
138:             $this->usingTz        = true;
139:             $this->externalOffset = Util::cleanTimeZoneOffset($externalOffset);
140:             $this->internalOffset = Util::cleanTimeZoneOffset($internalOffset);
141:         }
142:     }
143:     
144:     /**
145:      * デフォルトのインスタンスを返します.
146:      * このメソッドから生成されたオブジェクトは, parse の際にタイムゾーンを一切考慮しません.
147:      * また, 書式化する際にタイムゾーン文字列を付与しません.
148:      * 
149:      * @return W3cDatetimeFormat タイムゾーンに対応しないインスタンス
150:      * @codeCoverageIgnore
151:      */
152:     public static function getInstance()
153:     {
154:         static $instance = null;
155:         if ($instance === null) {
156:             $instance = new self(false);
157:         }
158:         return $instance;
159:     }

161:     /**
162:      * "YYYY-MM-DD" 形式の文字列を解析します.
163:      * 
164:      * @param  string $format 解析対象の文字列
165:      * @return Date           解析結果
166:      * @see Format::parseDate()
167:      */
168:     public function parseDate($format)
169:     {
170:         $dp = self::$datePattern;
171:         if (preg_match("/^{$dp}/", $format)) {
172:             $year  = substr($format, 0, 4);
173:             $month = substr($format, 5, 2);
174:             $date  = substr($format, 8, 2);
175:             return new Date($year, $month, $date);
176:         } else {
177:             $this->throwFormatException($format, "YYYY-MM-DD");
178:         }
179:     }
180:     
181:     /**
182:      * "YYYY-MM-DDThh:mm" 形式の文字列を解析します.
183:      * 文字列の末尾にタイムゾーン (+hh:mm や -hh:mm など) を含む場合は, システム時刻への変換を行います.
184:      * 
185:      * @param  string $format 解析対象の文字列
186:      * @return Datetime       解析結果
187:      * @see Format::parseDatetime()
188:      */
189:     public function parseDatetime($format)
190:     {
191:         $dp     = self::$datePattern;
192:         $tzp    = self::$timeZonePattern;
193:         $result = null;
194:         if (!preg_match("/^{$dp}[^0-9][0-9]{2}:[0-9]{2}{$tzp}/", $format, $result)) {
195:             $this->throwFormatException($format, "YYYY-MM-DD hh:mm[timezone]");
196:         }
197:         
198:         $year   = substr($format, 0,  4);
199:         $month  = substr($format, 5,  2);
200:         $date   = substr($format, 8,  2);
201:         $hour   = substr($format, 11, 2);
202:         $min    = substr($format, 14, 2);
203:         $obj    = new Datetime($year, $month, $date, $hour, $min);
204:         return ($this->usingTz && isset($result[1])) ?
205:             $this->adjustFromParse($obj, $result[1]) : $obj;
206:     }
207:     
208:     /**
209:      * "YYYY-MM-DDThh:mm:ss" 形式の文字列を解析します.
210:      * 
211:      * @param  string $format 解析対象の文字列
212:      * @return Timestamp      解析結果
213:      * @see Format::parseTimestamp()
214:      */
215:     public function parseTimestamp($format)
216:     {
217:         $dp     = self::$datePattern;
218:         $tzp    = self::$timeZonePattern;
219:         $result = null;
220:         if (!preg_match("/^{$dp}[^0-9][0-9]{2}:[0-9]{2}:[0-9]{2}{$tzp}/", $format, $result)) {
221:             $this->throwFormatException($format, "YYYY-MM-DD hh:mm:ss");
222:         }
223:         
224:         $year   = substr($format, 0,  4);
225:         $month  = substr($format, 5,  2);
226:         $date   = substr($format, 8,  2);
227:         $hour   = substr($format, 11, 2);
228:         $min    = substr($format, 14, 2);
229:         $sec    = substr($format, 17, 2);
230:         $obj    = new Timestamp($year, $month, $date, $hour, $min, $sec);
231:         return ($this->usingTz && isset($result[1])) ?
232:             $this->adjustFromParse($obj, $result[1]) : $obj;
233:     }
234:     
235:     /**
236:      * 指定された時間オブジェクトを "YYYY-MM-DD" 形式の文字列に変換します.
237:      * 
238:      * @param  Date $d 解析対象の Date オブジェクト
239:      * @return string  "YYYY-MM-DD" 形式の文字列
240:      * @see    Format::formatDate()
241:      */
242:     public function formatDate(Date $d)
243:     {
244:         return $d->__toString();
245:     }
246:     
247:     /**
248:      * 指定された時間オブジェクトを "YYYY-MM-DDThh:mm" 形式の文字列に変換します.
249:      * このインスタンスがタイムゾーンに対応している場合, 末尾にタイムゾーン文字列も付加します.
250:      * 
251:      * @param  Datetime $d 変換対象の Datetime オブジェクト
252:      * @return string      "YYYY-MM-DDThh:mm" 形式の文字列
253:      * @see    Format::formatDatetime()
254:      */
255:     public function formatDatetime(Datetime $d)
256:     {
257:         $a = $this->adjustFromFormat($d, false);
258:         return $this->formatDate($a->toDate()) . "T" . $a->formatTime() . $this->formatTimezone();
259:     }
260:     
261:     /**
262:      * 指定された時間オブジェクトを "YYYY-MM-DDThh:mm:ss" 形式の文字列に変換します.
263:      * このインスタンスがタイムゾーンに対応している場合, 末尾にタイムゾーン文字列も付加します.
264:      * 
265:      * @param  Timestamp $d 変換対象の Timestamp オブジェクト
266:      * @return string       "YYYY-MM-DDThh:mm:ss" 形式の文字列
267:      * @see    Format::formatTimestamp()
268:      */
269:     public function formatTimestamp(Timestamp $d)
270:     {
271:         return $this->formatDatetime($d);
272:     }
273:     
274:     /**
275:      * 指定された文字列が想定されたフォーマットでなかった際に呼ばれるメソッドです.
276:      * 
277:      * @param string $format
278:      * @param string $expected
279:      * @throws \InvalidArgumentException
280:      */
281:     private function throwFormatException($format, $expected)
282:     {
283:         throw new \InvalidArgumentException("Illegal format({$format}). Expected: {$expected}");
284:     }
285:     
286:     /**
287:      * タイムゾーン文字列をパースして, 時差を分単位で返します.
288:      * NULL が指定された場合 (タイムゾーン文字列がなかった場合),
289:      * このオブジェクトに設定されている externalOffset の値を返します.
290:      * 
291:      * @param  string $tz NULL, "Z", または "+h:mm", "-h:mm" 形式の文字列
292:      * @return int
293:      */
294:     private function parseTimezone($tz)
295:     {
296:         if ($tz === "Z") {
297:             return 0;
298:         }
299:         
300:         $coronIndex = strpos($tz, ":");
301:         $sign       = substr($tz, 0, 1);
302:         $hour       = substr($tz, 1, $coronIndex - 1);
303:         $minute     = substr($tz, $coronIndex + 1);
304:         $offset     = $hour * 60 + $minute;
305:         return ($sign === "-") ? $offset : - $offset;
306:     }
307:     
308:     /**
309:      * フォーマットのタイムゾーンと, 時間オブジェクトのタイムゾーンを元に
310:      * 指定された時間オブジェクトの補正処理を行います.
311:      * 
312:      * @param  Datetime $d  補正対象の時間オブジェクト
313:      * @param  string   $tz タイムゾーン文字列
314:      * @return Datetime     補正結果の時間オブジェクト
315:      */
316:     private function adjustFromParse(Datetime $d, $tz)
317:     {
318:         $externalOffset = $this->parseTimezone($tz);
319:         return $d->add("minute", $externalOffset - $this->internalOffset);
320:     }
321:     
322:     /**
323:      * このオブジェクトがタイムゾーンを利用する場合は,
324:      * 引数の時間オブジェクトを補正した結果を返します.
325:      * タイムゾーンを利用しない場合は引数をそのまま返します.
326:      * このメソッドは各種 format メソッドから呼ばれます.
327:      * 
328:      * @param  Datetime $d 補正対象の時間オブジェクト
329:      * @return Datetime    補正結果
330:      */
331:     private function adjustFromFormat(Datetime $d)
332:     {
333:         return $this->usingTz ? $d->add("minute", $this->internalOffset - $this->externalOffset) : $d;
334:     }
335:     
336:     /**
337:      * タイムゾーンを書式化します.
338:      * @return string
339:      */
340:     private function formatTimezone()
341:     {
342:         if (!$this->usingTz) {
343:             return "";
344:         }
345:         if ($this->externalOffset === 0) {
346:             return "Z";
347:         }
348:         
349:         $ext    = $this->externalOffset;
350:         $format = (0 < $ext) ? "-" : "+";
351:         $offset = abs($ext);
352:         $hour   = intval($offset / 60);
353:         $minute = intval($offset % 60);
354:         $format .= str_pad($hour,   2, '0', STR_PAD_LEFT);
355:         $format .= ':';
356:         $format .= str_pad($minute, 2, '0', STR_PAD_LEFT);
357:         return $format;
358:     }
