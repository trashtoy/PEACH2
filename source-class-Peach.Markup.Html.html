241:     /**
242:      * select 要素に含まれる option の一覧を作成します.
243:      *

246:      * @return NodeList option 要素の一覧

248:     private static function createOptions($current, array $candidates)
249:     {
250:         $result = new NodeList();
251:         foreach ($candidates as $key => $value) {
252:             if (is_array($value)) {
253:                 $optgroup = new ContainerElement("optgroup");
254:                 $optgroup->setAttribute("label", $key);
255:                 $optgroup->append(self::createOptions($current, $value));
256:                 $result->append($optgroup);
257:                 continue;
258:             }
259:             
260:             $option  = new ContainerElement("option");
261:             $option->setAttribute("value", $value);
262:             $value   = Values::stringValue($value);
263:             if ($current === $value) {
264:                 $option->setAttribute("selected");
265:             }
266:             $option->append($key);
267:             $result->append($option);
268:         }
269:         return $result;

289:     /**
290:      * このクラスで定義されているメソッドを簡単に呼び出すためのエイリアスを定義します.
291:      * 引数の配列では, キーにメソッド名, 値にそのメソッドのエイリアス (関数名) を指定してください.
292:      * キーに使用することが出来る文字列 (メソッド名) は以下の通りです.
293:      * 
294:      * - {@link Html::tag()}
295:      * - {@link Html::comment()}
296:      * - {@link Html::conditionalComment()}
297:      * - {@link Html::select()}
298:      * 
299:      * 使用例を挙げます.
300:      * <code>
301:      * Html::alias(array("tag" => "t", "comment" => "co"));
302:      * echo t("div")
303:      *     ->append(co("TEST"))
304:      *     ->append(t("h1")->append("Sample"))
305:      *     ->append(t("p")->append("Hello World!"))
306:      *     ->write();
307:      * </code>
308:      * このコードは以下の内容と等価です.
309:      * <code>
310:      * echo Html::tag("div")
311:      *     ->append(Html::comment("TEST"))
312:      *     ->append(Html::tag("h1")->append("Sample"))
313:      *     ->append(Html::tag("p")->append("Hello World!"))
314:      *     ->write();
315:      * </code>
316:      * いずれも以下の結果を出力します.
317:      * <code>
318:      * <div>
319:      *     <!--TEST-->
320:      *     <h1>Sample</h1>
321:      *     <p>Hello world!</p>
322:      * </div>
323:      * </code>
324:      * 
325:      * 引数を省略した場合は array("tag" => "tag") が指定されたものとして扱います.
326:      * ({@link Html::tag()} のエイリアスとして関数 tag() が定義されます)
327:      * 
328:      * 同じメソッドと別名の組み合わせで alias を複数回実行した場合,
329:      * 二度目以降のものは無視されます.
330:      * 
331:      * @param  array $options 定義するエイリアスの一覧. キーがメソッド, 値が新しく定義する関数名
332:      * @throws InvalidArgumentException 不適切なクラスメソッド名を指定した /
333:      * 指定された関数名が既に存在していた / 関数名が不適切など
334:      */
335:     public static function alias(array $options = array())
336:     {
337:         if (!count($options)) {
338:             return self::alias(array("tag" => "tag"));
339:         }
340:         
341:         static $definedList = array();
342:         $newOptions = array();
343:         foreach ($options as $methodName => $funcName) {
344:             if (!isset($definedList[$funcName]) || $definedList[$funcName] !== $methodName) {
345:                 $newOptions[$methodName] = $funcName;
346:                 $definedList[$funcName]  = $methodName;
347:             }
348:         }
349:         self::handleAlias($newOptions);
350:     }

352:     /**
353:      * alias() の本処理です.
354:      * @param  array $options
355:      * @throws InvalidArgumentException
356:      */
357:     private static function handleAlias(array $options)
358:     {
359:         static $validNames = array("tag", "comment", "conditionalComment", "select");
360:         foreach ($options as $methodName => $funcName) {
361:             if (!in_array($methodName, $validNames, true)) {
362:                 throw new \InvalidArgumentException("Method '{$methodName}' is not found.");
363:             }
364:             if (function_exists($funcName)) {
365:                 throw new \InvalidArgumentException("Function '{$funcName}' is already defined.");
366:             }
367:             // http://www.php.net/manual/ja/functions.user-defined.php を参考
368:             if (!preg_match("/^[a-zA-Z_\x7f-\xff][a-zA-Z0-9_\x7f-\xff]*$/", $funcName)) {
369:                 throw new \InvalidArgumentException("Invalid function name: '{$funcName}'");
370:             }
371:         }
372:         
373:         if (isset($options["tag"])) {
374:             $tag = $options["tag"];
375:             eval("function {$tag}(\$n = null, \$a = array()) { return \Peach\Markup\Html::tag(\$n, \$a); }");
376:         }
377:         if (isset($options["comment"])) {
378:             $comment = $options["comment"];
379:             eval("function {$comment}(\$c, \$p = '', \$s = '') { return \Peach\Markup\Html::comment(\$c, \$p, \$s); }");
380:         }
381:         if (isset($options["conditionalComment"])) {
382:             $cond = $options["conditionalComment"];
383:             eval("function {$cond}(\$c1, \$c2) { return \Peach\Markup\Html::conditionalComment(\$c1, \$c2); }");
384:         }
385:         if (isset($options["select"])) {
386:             $select = $options["select"];
387:             eval("function {$select}(\$c1, \$c2, \$a = array()) { return \Peach\Markup\Html::select(\$c1, \$c2, \$a); }");
388:         }
389:     }

391:     /**
392:      * このクラスで定義されているメソッドを呼び出すためのクロージャを生成します.
393:      * 引数に指定することができるメソッド名は以下の通りです.
394:      * 
395:      * - {@link Html::tag()}
396:      * - {@link Html::comment()}
397:      * - {@link Html::conditionalComment()}
398:      * - {@link Html::select()}
399:      * 
400:      * 使用例を挙げます.
401:      * <code>
402:      * $t = Html::closure("tag");
403:      * $c = Html::closure("comment");
404:      * echo $t("div")
405:      *     ->append($c("TEST"))
406:      *     ->append($t("h1")->append("Sample"))
407:      *     ->append($t("p")->append("Hello World!"))
408:      *     ->write();
409:      * </code>
410:      * このコードは以下の内容と等価です.
411:      * <code>
412:      * echo Html::tag("div")
413:      *     ->append(Html::comment("TEST"))
414:      *     ->append(Html::tag("h1")->append("Sample"))
415:      *     ->append(Html::tag("p")->append("Hello World!"))
416:      *     ->write();
417:      * </code>
418:      * いずれも以下の結果を出力します.
419:      * <code>
420:      * <div>
421:      *     <!--TEST-->
422:      *     <h1>Sample</h1>
423:      *     <p>Hello world!</p>
424:      * </div>
425:      * </code>
426:      * 
427:      * 同様の処理を行うメソッドとして alias() がありますが,
428:      * alias() は関数定義を行うことでグローバルスコープに影響を及ぼしてしまうため,
429:      * 使用することはあまり好ましくありません.
430:      * 特に理由がない限り alias() ではなく closure() を使用するようにしてください.
431:      * 
432:      * @param string $name メソッド名
433:      * @return \Closure 指定されたメソッド名を呼び出すクロージャ
434:      * @throws \InvalidArgumentException 指定されたメソッド名が存在しない場合
435:      */
436:     public static function closure($name)
437:     {
438:         static $closures = array("tag" => null, "comment" => null, "conditionalComment" => null, "select" => null);
439:         if (!array_key_exists($name, $closures)) {
440:             throw new \InvalidArgumentException("Method '{$name}' is not found.");
441:         }
442:         if ($closures[$name] === null) {
443:             $closures[$name] = self::createClosure($name);
444:         }
445:         return $closures[$name];
446:     }
447:     
448:     /**
449:      * 指定されたメソッドのクロージャを生成します.
450:      * @param string $name メソッド名
451:      * @return \Closure
452:      */
453:     private static function createClosure($name)
454:     {
455:         switch ($name) {
456:             case "tag":
457:                 return function ($n = null, $a = array()) { return self::tag($n, $a); };
458:             case "comment":
459:                 return function ($c, $p = "", $s = "") { return self::comment($c, $p, $s); };
460:             case "conditionalComment":
461:                 return function ($c1, $c2) { return self::conditionalComment($c1, $c2); };
462:             case "select":
463:                 return function ($c1, $c2, $a = array()) { return self::select($c1, $c2, $a); };
464:         }
465:         
466:         // @codeCoverageIgnoreStart
467:         throw new \Excption("Invalid name specified.");
468:         // @codeCoverageIgnoreEnd
469:     }
470:     
471:     /**
472:      * 指定されたメソッド名を呼び出すクロージャの一覧を返します.
473:      * 使用例を挙げます.
474:      * <code>
475:      * list($t, $c) = Html::closures(array("tag", "comment"));
476:      * </code>
477:      * このコードは以下の内容と等価です.
478:      * <code>
479:      * $t = Html::closure("tag");
480:      * $c = Html::closure("comment");
481:      * </code>
482:      * 
483:      * 以下のように, メソッド名をキーとして対応するクロージャを取り出すこともできます.
484:      * <code>
485:      * $closures = Html::closures(array("tag", "comment"));
486:      * $t = $closures["tag"];
487:      * $c = $closures["comment"];
488:      * </code>
489:      * 
490:      * @param  array $names メソッド名の配列
491:      * @return array        指定されたメソッドを呼び出すクロージャの配列
492:      * @throws \InvalidArgumentException 配列内に不正なメソッド名が 1 つ以上存在した場合
493:      */
494:     public static function closures(array $names)
495:     {
496:         $result = array();
497:         foreach ($names as $name) {
498:             $name          = Values::stringValue($name);
499:             $closure       = self::closure($name);
500:             $result[]      = $closure;
501:             $result[$name] = $closure;
502:         }
503:         return $result;
504:     }
