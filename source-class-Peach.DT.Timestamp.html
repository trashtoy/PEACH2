 49:     {
 50:         $time  = time();
 51:         $year  = date("Y", $time);
 52:         $month = date("n", $time);
 53:         $date  = date("d", $time);
 54:         $hour  = date("H", $time);
 55:         $min   = date("i", $time);

 57:         return new self($year, $month, $date, $hour, $min, $sec);
 58:     }

 60:     /**
 61:      * 指定されたテキストを解析して Timestamp オブジェクトに変換します.

 63:      * を使って解析を行います.
 64:      * ("YYYY-MM-DD hh:mm:ss" 形式の文字列を受理します.

 66:      *

121:      * 
122:      * @param  Time $time 比較対象の時間

124:      * @ignore

126:     protected function compareFields(Time $time)
127:     {
128:         $c = parent::compareFields($time);
129:         if (!isset($c) || $c != 0) {

132:         $className = __CLASS__;

134:             return $this->second - $time->second;
