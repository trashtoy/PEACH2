171:      * この時間と指定された時間を比較します.
172:      * 
173:      * この型の時間フィールドと引数の型の時間フィールドのうち,
174:      * 共通しているフィールド同士を比較します.
175:      * 
176:      * 引数がこのクラスを継承したオブジェクトではない場合,
177:      * 引数のオブジェクトに対して get("year"), get("month"), get("date"), get("hour"), get("minute")
178:      * を呼び出した結果を比較対象のフィールドとします.
179:      * 
180:      * @param  Time 比較対象の時間
181:      * @return int  この時間のほうが過去の場合は負の値, 未来の場合は正の値, それ以外は 0
182:      * @ignore
183:      */
184:     protected function compareFields(Time $time)
185:     {
186:         $c = parent::compareFields($time);
187:         if ($c !== 0) {
188:             return $c;
189:         }
190:         
191:         $className = __CLASS__;
192:         if ($time instanceof $className) {
193:             if ($this->hour   !== $time->hour) {
194:                 return $this->hour   - $time->hour;
195:             }
196:             if ($this->minute !== $time->minute) {
197:                 return $this->minute - $time->minute;
198:             }
199:             return 0;
200:         }
201:         else {
202:             $h = $time->get("hour");
203:             $m = $time->get("minute");
204:             if ($this->hour   !== $h) {
205:                 return (isset($h) ? $this->hour   - $h : 0);
206:             }
207:             if ($this->minute !== $m) {
208:                 return (isset($m) ? $this->minute - $m : 0);
209:             }
210:             return 0;
211:         }
212:     }
